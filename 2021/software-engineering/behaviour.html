enum Light {
  Red, Orange, Green
};
class TrafficLight {
  Light state;
  void updateState() {
    switch(state) {
      case Red:
        state = Green;
        break;
      case Orange:
        state = Red;
        break;
      case Green:
        state = Orange;
        break;
    }
  }
}
